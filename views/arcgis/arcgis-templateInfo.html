           stationPointLayer.on('mouse-out',function (e) {
                templateInfoDel(e);
            });
            function templateInfoInit(e) {
                var geometry = map.toScreen(e.graphic.geometry);
                var div = dojo.doc.createElement("div");
                dojo.attr(div, {
                    id: "text"
                });
                dojo.style(div, {
                    left: geometry.x + 10 + "px",
                    top: geometry.y + 10 + "px",
                    position: "absolute",
                    "z-index": 99,
                    background: "#fff",
                    border: "1px solid #0096ff",
                    padding: "4px 8px 4px",
                    "font-size": "11px",
                    "border-radius": "3px",
                    "box-shadow": "0 0 0.75em #777777"
                });
                div.innerHTML = "名称：" + e.graphic.attributes.name + " <br>ID    ：" + e.graphic.attributes.id + " <br>状态：" + e.graphic.attributes.status + " <br> ";
                dom.byId("map").appendChild(div);
            }
